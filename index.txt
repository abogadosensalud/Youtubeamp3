    <script>
        const API_BASE_URL = 'https://tu-app.railway.app'; // Cambia esto por tu URL de Railway
        
        let selectedFormat = 'mp3';

        // Manejo de selecciÃ³n de formato
        document.querySelectorAll('.format-btn').forEach(btn => {
            btn.addEventListener('click', () => {
                document.querySelectorAll('.format-btn').forEach(b => b.classList.remove('active'));
                btn.classList.add('active');
                selectedFormat = btn.dataset.format;
            });
        });

        // FunciÃ³n para mostrar estado
        function showStatus(message, type) {
            const statusDiv = document.getElementById('statusDiv');
            statusDiv.className = `status ${type}`;
            statusDiv.style.display = 'block';
            
            if (type === 'loading') {
                statusDiv.innerHTML = `<div class="loading-spinner"></div>${message}`;
            } else {
                statusDiv.innerHTML = message;
            }
        }

        // FunciÃ³n para validar URL de YouTube
        function isValidYouTubeUrl(url) {
            const regex = /^(https?:\/\/)?(www\.)?(youtube\.com|youtu\.be)\/.+/;
            return regex.test(url);
        }

        // FunciÃ³n principal de descarga
        async function downloadContent() {
            const url = document.getElementById('urlInput').value.trim();
            const downloadBtn = document.getElementById('downloadBtn');

            if (!url) {
                showStatus('Por favor, ingresa una URL de YouTube', 'error');
                return;
            }

            if (!isValidYouTubeUrl(url)) {
                showStatus('Por favor, ingresa una URL vÃ¡lida de YouTube', 'error');
                return;
            }

            downloadBtn.disabled = true;
            showStatus('Procesando tu solicitud...', 'loading');

            try {
                const response = await fetch(`${API_BASE_URL}/download`, {
                    method: 'POST',
                    headers: {
                        'Content-Type': 'application/json',
                    },
                    body: JSON.stringify({
                        url: url,
                        format: selectedFormat
                    })
                });

                if (!response.ok) {
                    throw new Error(`Error del servidor: ${response.status}`);
                }

                const data = await response.json();

                if (data.success) {
                    showStatus('Â¡Descarga iniciada! El archivo se descargarÃ¡ automÃ¡ticamente.', 'success');
                    
                    // Crear enlace de descarga
                    const link = document.createElement('a');
                    link.href = data.downloadUrl;
                    link.download = data.filename;
                    link.click();
                } else {
                    showStatus(data.error || 'Error al procesar la descarga', 'error');
                }

            } catch (error) {
                console.error('Error:', error);
                showStatus('Error de conexiÃ³n. Por favor, intenta nuevamente.', 'error');
            } finally {
                downloadBtn.disabled = false;
            }
        }

        // Event listeners
        document.getElementById('downloadBtn').addEventListener('click', downloadContent);

        document.getElementById('urlInput').addEventListener('keypress', (e) => {
            if (e.key === 'Enter') {
                downloadContent();
            }
        });

        // Limpiar estado cuando se cambia la URL
        document.getElementById('urlInput').addEventListener('input', () => {
            document.getElementById('statusDiv').style.display = 'none';
        });
